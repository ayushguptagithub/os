              #include<sys/types.h>
              #include<sys/stat.h>
              #include<fcntl.h>
              #include<stdio.h>
              int main() {
                  int n, fd;
                  char buff[50];
                  fd = open("test.txt", O_RDONLY); 
                  printf("The file descriptor of the file is: %d\n", fd);
                  n = read(fd, buff, 10);
                  write(1, buff, n);
              }

Program 1

                1. Start
                2. Include necessary header files: unistd.h, sys/types.h, sys/stat.h, fcntl.h, stdio.h
                3. Define the main function
                    a. Declare variables: n (for read/write length), fd (for file descriptor), buff[50] (for data)
                    b. Open the file "test.txt" in read-only mode using the open() function and save the file descriptor in fd
                    c. Print the file descriptor value using printf()
                    d. Read 10 characters from the file pointed to by fd and store them in buffer (buff) using the read() function
                    e. Write the data from buffer to standard output (screen) using the write() function
                4. End main function
