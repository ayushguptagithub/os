Aim: WAP to create proccess in linux using fork system

                #include<stdio.h>
                #include<unistd.h>
                #include<sys/types.h>
                int main()
                {
                    pid_t p;
                    printf("before fork\n");
                    p = fork();
                    if (p == 0)
                    {
                        printf("I am child having id %d\n", getpid());
                        printf("My parent's id is %d\n", getppid());
                    }
                    else
                    {
                        printf("My child's id is %d\n", p);
                        printf("I am parent having id %d\n", getpid());
                    }
                    printf("Common\n");
                }
            
                1. Start
                2. Include necessary header files: stdio.h, unistd.h, sys/types.h
                3. Define the main function
                    a. Declare a variable to hold process ID (pid_t p)
                    b. Print "before fork" message
                    c. Call fork() function to create a child process and store its ID in variable p
                    d. Check if the return value of fork() is 0 (indicating child process)
                        i. Print child process ID using getpid() function
                        ii. Print parent process ID using getppid() function
                    e. If return value of fork() is not 0 (indicating parent process)
                        i. Print child process ID stored in variable p
                        ii. Print parent process ID using getpid() function
                    f. Print "Common" message (executed by both parent and child)
                4. End main function
